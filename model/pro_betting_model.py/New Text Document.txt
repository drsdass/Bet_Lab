from __future__ import annotations

from dataclasses import dataclass, asdict
from typing import Optional


@dataclass
class GameInput:
    date: str
    league: str
    team_a: str
    team_b: str
    spread: float
    total: float
    proj_spread: float
    proj_total: float
    pace: float
    off_eff_a: float
    def_eff_a: float
    off_eff_b: float
    def_eff_b: float
    opening_spread: Optional[float] = None
    closing_spread: Optional[float] = None
    opening_total: Optional[float] = None
    closing_total: Optional[float] = None
    public_pct_team_a: Optional[float] = None
    public_pct_team_b: Optional[float] = None
    public_pct_over: Optional[float] = None
    public_pct_under: Optional[float] = None
    is_back_to_back: bool = False
    travel_diff: int = 0
    home_advantage: bool = True
    star_out: bool = False


def projection_edge(game: GameInput) -> float:
    return abs(game.proj_spread - game.spread)


def total_edge(game: GameInput) -> float:
    return abs(game.proj_total - game.total)


def clv_signal(game: GameInput) -> float:
    if game.opening_spread is None or game.closing_spread is None:
        return 0.0
    return game.opening_spread - game.closing_spread


def reverse_line_movement(game: GameInput) -> bool:
    if (
        game.opening_spread is None
        or game.closing_spread is None
        or game.public_pct_team_a is None
        or game.public_pct_team_b is None
    ):
        return False

    public_on_a = game.public_pct_team_a >= 65
    public_on_b = game.public_pct_team_b >= 65

    # Line moving toward B means closing spread is less favorable to A.
    move_toward_b = game.closing_spread > game.opening_spread
    move_toward_a = game.closing_spread < game.opening_spread

    return (public_on_a and move_toward_b) or (public_on_b and move_toward_a)


def market_signal(game: GameInput) -> float:
    score = 0.0
    clv = clv_signal(game)
    if abs(clv) >= 1.0:
        score += 2.0
    if reverse_line_movement(game):
        score += 2.0
    return score


def matchup_edge(game: GameInput) -> float:
    edge_a = game.off_eff_a - game.def_eff_b
    edge_b = game.off_eff_b - game.def_eff_a
    return (edge_a - edge_b) / 10.0


def situational_score(game: GameInput) -> float:
    score = 0.0
    if game.is_back_to_back:
        score -= 2.0
    if game.travel_diff > 0:
        score -= 1.0
    if game.home_advantage:
        score += 1.0
    return score


def player_impact(game: GameInput) -> float:
    return -3.0 if game.star_out else 1.0


def volatility_adjustment(game: GameInput) -> float:
    if abs(game.spread) >= 10:
        return -2.0
    if abs(game.spread) in {8.5, 9.5, 10.5}:
        return -1.5
    if abs(game.spread) <= 6:
        return 1.0
    return 0.0


def final_edge_score(game: GameInput) -> float:
    return (
        0.30 * projection_edge(game)
        + 0.20 * market_signal(game)
        + 0.15 * matchup_edge(game)
        + 0.15 * situational_score(game)
        + 0.10 * player_impact(game)
        + 0.10 * volatility_adjustment(game)
    )


def best_bet(game: GameInput) -> str:
    spread_pick = game.team_a if game.proj_spread < game.spread else game.team_b
    spread_line = game.spread if spread_pick == game.team_a else -game.spread

    if total_edge(game) > projection_edge(game):
        if game.proj_total > game.total:
            return f"Over {game.total}"
        return f"Under {game.total}"

    return f"{spread_pick} {spread_line:+}"


def classify_tier(score: float) -> str:
    if score >= 8:
        return "MAX_ELITE"
    if score >= 6:
        return "ELITE"
    if score >= 4.5:
        return "STRONG"
    return "PASS"


def build_signals(game: GameInput) -> list[str]:
    signals: list[str] = []
    if abs(clv_signal(game)) >= 1.0:
        signals.append("CLV")
    if reverse_line_movement(game):
        signals.append("RLM")
    if abs(game.spread) in {8.5, 9.5, 10.5}:
        signals.append("HOOK_RISK")
    if abs(game.spread) >= 10:
        signals.append("BLOWOUT_RISK")
    if game.is_back_to_back:
        signals.append("B2B")
    if game.star_out:
        signals.append("STAR_OUT")
    if total_edge(game) > projection_edge(game):
        signals.append("TOTAL_EDGE")
    else:
        signals.append("SPREAD_EDGE")
    return signals


def evaluate_game(game: GameInput) -> dict:
    score = round(final_edge_score(game), 2)
    return {
        "date": game.date,
        "league": game.league,
        "game": f"{game.team_a} vs {game.team_b}",
        "best_bet": best_bet(game),
        "spread": game.spread,
        "total": game.total,
        "proj_spread": round(game.proj_spread, 2),
        "proj_total": round(game.proj_total, 2),
        "edge_pct": round(projection_edge(game) * 100 / 10, 1),
        "win_prob": round(min(75, 50 + score * 2), 1),
        "score": score,
        "tier": classify_tier(score),
        "signals": build_signals(game),
    }
